7 // ****************************************************************************
8 //
9 // Purpose:
10 //
11 // JACOBI_POLY evaluates the Jacobi polynomials at X.
12 //
13 // Differential equation:
14 //
15 // (1-X*X) Y'' + (BETA-ALPHA-(ALPHA+BETA+2) X) Y' + N (N+ALPHA+BETA+1) Y = 0
16 //
17 // Recursion:
18 //
19 // P(0,ALPHA,BETA,X) = 1,
20 //
21 // P(1,ALPHA,BETA,X) = ( (2+ALPHA+BETA)*X + (ALPHA-BETA) ) / 2
22 //
23 // P(N,ALPHA,BETA,X) =
24 // (
25 // (2*N+ALPHA+BETA-1)
26 // * ((ALPHA**2-BETA**2)+(2*N+ALPHA+BETA)*(2*N+ALPHA+BETA-2)*X)
27 // * P(N-1,ALPHA,BETA,X)
28 // -2*(N-1+ALPHA)*(N-1+BETA)*(2*N+ALPHA+BETA) * P(N-2,ALPHA,BETA,X)
29 // ) / 2*N*(N+ALPHA+BETA)*(2*N-2+ALPHA+BETA)
30 //
31 // Restrictions:
32 //
33 // -1 < ALPHA
34 // -1 < BETA
35 //
36 // Norm:
37 //
38 // Integral ( -1 <= X <= 1 ) ( 1 - X )**ALPHA * ( 1 + X )**BETA
39 // * P(N,ALPHA,BETA,X)**2 dX
40 // = 2**(ALPHA+BETA+1) * Gamma ( N + ALPHA + 1 ) * Gamma ( N + BETA + 1 ) /
41 // ( 2 * N + ALPHA + BETA ) * N! * Gamma ( N + ALPHA + BETA + 1 )
42 //
43 // Special values:
44 //
45 // P(N,ALPHA,BETA)(1) = (N+ALPHA)!/(N!*ALPHA!) for integer ALPHA.
46 //
47 // Modified:
48 //
49 // 12 May 2003
50 //
51 // Author:
52 //
53 // John Burkardt
54 //
55 // Reference:
56 //
57 // Milton Abramowitz and Irene Stegun,
58 // Handbook of Mathematical Functions,
59 // US Department of Commerce, 1964.
60 //
61 // Parameters:
62 //
63 // Input, int N, the highest order polynomial to compute. Note
64 // that polynomials 0 through N will be computed.
65 //
66 // Input, double ALPHA, one of the parameters defining the Jacobi
67 // polynomials, ALPHA must be greater than -1.
68 //
69 // Input, double BETA, the second parameter defining the Jacobi
70 // polynomials, BETA must be greater than -1.
71 //
72 // Input, double X, the point at which the polynomials are to be evaluated.
73 //
74 // Output, double CX[N+1], the values of the first N+1 Jacobi
75 // polynomials at the point X.
76 //
77 {
78  double c1;
79  double c2;
80  double c3;
81  double c4;
82  int i;
83 
84  if ( alpha <= -1.0 )
85  {
86  cout << "\n";
87  cout << "JACOBI_POLY - Fatal error!\n";
88  cout << " Illegal input value of ALPHA = " << alpha << "\n";
89  cout << " But ALPHA must be greater than -1.\n";
90  exit ( 1 );
91  }
92 
93  if ( beta <= -1.0 )
94  {
95  cout << "\n";
96  cout << "JACOBI_POLY - Fatal error!\n";
97  cout << " Illegal input value of BETA = " << beta << "\n";
98  cout << " But BETA must be greater than -1.\n";
99  exit ( 1 );
100  }
101 
102  if ( n < 0 )
103  {
104  return;
105  }
106 
107  cx[0] = 1.0;
108 
109  if ( n == 0 )
110  {
111  return;
112  }
113 
114  cx[1] = ( 1.0 + 0.5 * ( alpha + beta ) ) * x
115  + 0.5 * ( alpha - beta );
116 
117  for ( i = 2; i <= n; i++ )
118  {
119  c1 = 2.0 * ( double ) ( i ) * ( ( double ) ( i ) + alpha + beta )
120  * ( ( double ) ( 2 * i - 2 ) + alpha + beta );
121 
122  c2 = ( ( double ) ( 2 * i - 1 ) + alpha + beta )
123  * ( ( double ) ( 2 * i ) + alpha + beta )
124  * ( ( double ) ( 2 * i - 2 ) + alpha + beta );
125 
126  c3 = ( ( double ) ( 2 * i - 1 ) + alpha + beta )
127  * ( alpha + beta ) * ( alpha - beta );
128 
129  c4 = - ( double ) ( 2 ) * ( ( double ) ( i - 1 ) + alpha )
130  * ( ( double ) ( i - 1 ) + beta )
131  * ( ( double ) ( 2 * i ) + alpha + beta );
132 
133  cx[i] = ( ( c3 + c2 * x ) * cx[i-1] + c4 * cx[i-2] ) / c1;
134 
135  }
136 
137  return;
138 }

148  double c0, c1, c2, c3, c4, ab, di, ym, yp, dym, dyp, d2ym, d2yp;

150  y = 1.0;

152  d2y = 0.0;
153  if (n == 0) return;
154 
155  ab = a + b;
156  y = (ab + 2.0) * 0.5 * x + (a - b) * 0.5;
157  dy = (ab + 2.0) * 0.5;
158  d2y = 0.0;
159  if (n == 1) return;
160

162  dyp = 0.0;
163  d2yp = 0.0;
164  for (int i = 2; i <= n; ++i) {
165  di = (double) i;
166  c0 = di * 2.0 + ab;
167  c1 = di * 2.0 * (di + ab) * (c0 - 2.0);
168  c2 = (c0 - 1.0) * (c0 - 2.0) * c0;
169  c3 = (c0 - 1.0) * (a - b) * ab;
170  c4 = (di + a - 1.0) * 2.0 * c0 * (di + b - 1.0);
171  ym = y;
172  y = ((c2 * x + c3) * y - c4 * yp) / c1;
173  yp = ym;
174  dym = dy;
175  dy = ((c2 * x + c3) * dy - c4 * dyp + c2 * yp) / c1;
176  dyp = dym;
177  d2ym = d2y;
178  d2y = ((c2 * x + c3) * d2y - c4 * d2yp + c2 * 2.0 * dyp) / c1;
179  d2yp = d2ym;
180  }

275 // **********************************************************************
276 //
277 // Purpose:
278 //
279 // Calculate the m roots of the Jacobi polynomial
280 // P(m,alpha,beta)
281 //
282 // *******************************************************************
283 {
284  double r,y,dy;
285  double dm= m;
286  double s, delta;
287  const double pi = 3.1415926535897932384626433832795;
288  for(int k=0; k < m; k++){
289  r=-cos((2.0*(double) k + 1.0)*pi/2.0/dm);
290  if(k>0) r=(r+x[k-1])/2.0;
291  do{
292  s=0;
293  for(int i=0;i<k;i++) s+=1.0/(r-x[i]);
294  Jacobi_P(m, alpha, beta,r,y,dy);
295  delta=-y/(dy-y*s);
296  r+=delta;
297  }while(delta>1.0e-10);
298  x[k]=r;
299  }
